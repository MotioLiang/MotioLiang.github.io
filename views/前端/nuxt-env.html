nuxt 配置环境变量

代码日记 nuxt

1.在根目录下新建env.js文件

export default {
  dev: {
    MODE: 'develpment',
    ENV_API: 'http://jsonplaceholder.typicode.com'
  },
  pro: {
    MODE: 'production',
    ENV_API: 'http://jsonplaceholder.typicode.com/110'
  }
}
